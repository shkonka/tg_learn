git init	Инициализация нового репозитория
git status	Статус файлов
git add <file>	Добавить файл в индекс
git commit -m "..."	Сделать коммит
git log	История коммитов
git diff	Посмотреть разницу в изменённых файлах
git push / git pull	Отправка/получение изменений на/с сервера
git remote add origin URL	Привязка к удалённому репозиторию
git remote -v проверка привязки к гитхабу
